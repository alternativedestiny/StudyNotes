1. 两数之和

   ```c++
   // 暴力解法
   // sizeof会造成溢出访问数组的问题，size则没有问题
   class Solution {
    public:
        vector<int> twoSum(vector<int>& nums, int target) {
            int l = nums.size();
            int o1,o2;
            int flag=0;
            for(int i=0; i<l; i++){
                for(int j=i+1; j<l; j++){
                    if(nums[i]+nums[j]==target){
                        o1=i;
                        o2=j;
                        flag = 1;
                        break;
                    }
                }
                if(flag==1){
                    break;
                }
            }
            return {o1,o2};
        }
    };
   ```
